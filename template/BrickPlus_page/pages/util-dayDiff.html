{# page: 日期天数间隔 #}
  • 计算两个日期相差天数,dayDiff(start, end)
参数说明
属性 说明 类型 默认值
start 开始时间，必须符合日期格式xxxx-xx-xx String
end 开始时间，必须符合日期格式xxxx-xx-xx String

实例


0天 
          
$(function() {
  var $start = $('#demo-day-diff-start'),
      $end = $('#demo-day-diff-end'),
      $text = $('#demo-day-diff'),
      dayDiff = BrickPlus.Util.dayDiff;

  var calcDays = function(a, b) {
    var days = dayDiff(a, b);
    $text.html(days)
  };

  $start.on('choose.ui.datetimepicker', function(e, date) {
    calcDays(date, $end.val())
  });

  $end.on('choose.ui.datetimepicker', function(e, date) {
    calcDays(date, $start.val())
  })
})
