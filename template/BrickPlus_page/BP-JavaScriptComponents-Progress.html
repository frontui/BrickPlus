$(function() {
    $('[data-percent]').on('click', function() {
      var percent = $(this).data('percent')
      $('#j-progress-1').progress('go', percent)
    })

    $('[data-status]').on('click', function() {
      var status = $(this).data('status')
      $('#j-progress-1').progress('change', status)
    })

    var striped = true
    $('[data-striped]').on('click', function() {
      striped = !striped
      $('#j-progress-1').progress('setStriped', striped)
    })

    $('#j-progress-1').on('completed.bp.progress', function() {
      alert('completed!')
    })
  })
